                class Pila {
                    constructor() {
                        this.almacenPila = {};  // Usamos un objeto para almacenar los elementos de la pila.
                        this.contPila = 0;      // Contador que mantiene el seguimiento de la cantidad de elementos en la pila.
                    }
                
                    // Método para agregar un nuevo elemento a la pila
                    push(elemento) {
                        this.almacenPila[this.contPila] = elemento;  // Almacenamos el elemento en la posición 'contPila'.
                        this.contPila++;  // Incrementamos el contador para el siguiente índice.
                        return this.almacenPila;  // Retornamos el objeto que contiene los elementos de la pila.
                    }
                
                    // Método para eliminar y retornar el último elemento de la pila
                    pop() {
                        if (this.contPila === 0) {  // Si la pila está vacía.
                            return null;  // Retornamos 'null' si no hay elementos.
                        }
                        this.contPila--;  // Decrementamos el contador de elementos.
                        const elemento = this.almacenPila[this.contPila];  // Obtenemos el último elemento.
                        delete this.almacenPila[this.contPila];  // Eliminamos el último elemento de la pila.
                        return elemento;  // Retornamos el elemento eliminado.
                    }
                
                    // Método para ver el último elemento sin eliminarlo
                    peek() {
                        if (this.contPila === 0) {  // Si la pila está vacía.
                            return null;  // Retornamos 'null' si no hay elementos.
                        }
                        return this.almacenPila[this.contPila - 1];  // Retornamos el último elemento sin eliminarlo.
                    }
                
                    // Método para obtener el tamaño de la pila
                    size() {
                        return this.contPila;  // El tamaño de la pila es igual al valor de 'contPila'.
                    }
                
                    // Método para imprimir el contenido de la pila
                    print() {
                        return this.almacenPila;  // Retornamos el objeto que contiene los elementos de la pila.
                    }
                }
